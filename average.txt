#include <stdio.h>
#include <string.h>

int main () {
	
    int  i,s;
    float sum = 0;
    printf("How many elements do you want to find for average\n");
    scanf("%d",&s);
    
    for(i =1 ; i <= s; i++) {
        int n;
    	printf("Enter number ");
    	scanf("%d", &n);
    	printf("%d", n);
    	sum = sum + n;
	}
	
	printf("\nsum: %f",sum/s);
    
    
	
	return 0;
}
